def get_counts(sequence):
    counts = {}
    for x in sequence:
        if x in counts:
            counts[x] += 1
        else:
            counts[x] = 1
    return counts

    value_key_pairs = [(count, tz) for tz, count in count_dict.items()]

    return value_key_pairs[-n:]
